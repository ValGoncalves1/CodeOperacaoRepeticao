class Program
{
    static void Main()
    {
        int soma = 0;

        // Loop para somar os números de 1 a 100
        for (int i = 1; i <= 100; i++)
        {
            soma += i;
        }

        // Exibe a soma total
        Console.WriteLine($"A soma dos números inteiros positivos de 1 a 100 é: {soma}");
    }
}
